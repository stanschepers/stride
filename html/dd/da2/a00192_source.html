63 void GeoGridConfig::SetData(const string& householdsFileName)
64 {
65  ReaderFactory readerFactory;
66 
67  auto householdsReader = readerFactory.CreateHouseholdReader(householdsFileName);
68  householdsReader->SetReferenceHouseholds(refHH.person_count, refHH.ages, refHH.young_old_fraction);
69  const auto popSize = param.pop_size;
70 
71  //----------------------------------------------------------------
72  // Determine age makeup of reference houshold population.
73  //----------------------------------------------------------------
74  const auto ref_p_count = refHH.person_count;
75  const auto averageHhSize = static_cast<double>(ref_p_count) / static_cast<double>(refHH.ages.size());
76 
77  auto ref_daycare = 0U;
78  auto ref_preschool = 0U;
79  auto ref_k12school = 0U;
80  auto ref_college = 0U;
81  auto ref_workplace = 0U;
82  for (const auto& hhAgeProfile : refHH.ages) {
83  for (const auto& age : hhAgeProfile) {
84  if (Daycare::HasAge(age)) {
85  ref_daycare++;
86  }
87  if (PreSchool::HasAge(age)) {
88  ref_preschool++;
89  }
90  if (K12School::HasAge(age)) {
91  ref_k12school++;
92  }
93  if (College::HasAge(age)) {
94  ref_college++;
95  }
96  if (Workplace::HasAge(age)) {
97  ref_workplace++;
98  }
99  }
100  }
101  //----------------------------------------------------------------
102  // Scale up to the generated population size.
103  //----------------------------------------------------------------
104  const auto fraction_daycare_age = static_cast<double>(ref_daycare) / static_cast<double>(ref_p_count);
105  const auto fraction_preschool_age = static_cast<double>(ref_preschool) / static_cast<double>(ref_p_count);
106  const auto fraction_k12school_age = static_cast<double>(ref_k12school) / static_cast<double>(ref_p_count);
107  const auto fraction_college_age = static_cast<double>(ref_college) / static_cast<double>(ref_p_count);
108  const auto fraction_workplace_age = static_cast<double>(ref_workplace) / static_cast<double>(ref_p_count);
109 
110  const auto age_count_daycare = static_cast<unsigned int>(floor(popSize * fraction_daycare_age));
111  const auto age_count_preschool = static_cast<unsigned int>(floor(popSize * fraction_preschool_age));
112  const auto age_count_k12school = static_cast<unsigned int>(floor(popSize * fraction_k12school_age));
113  const auto age_count_college = static_cast<unsigned int>(floor(popSize * fraction_college_age));
114  const auto age_count_workplace = static_cast<unsigned int>(floor(popSize * fraction_workplace_age));
115 
116  info.popcount_daycare = static_cast<unsigned int>(floor(param.participation_daycare * age_count_daycare));
117 
118  info.popcount_preschool = static_cast<unsigned int>(floor(param.participation_preschool * age_count_preschool));
119 
120  info.popcount_k12school = age_count_k12school;
121 
122  info.popcount_college = static_cast<unsigned int>(floor(param.participation_college * age_count_college));
123 
124  info.popcount_workplace = static_cast<unsigned int>(
125  floor(param.participation_workplace * (age_count_workplace - info.popcount_college)));
126 
127  info.count_households = static_cast<unsigned int>(floor(static_cast<double>(popSize) / averageHhSize));
128 }
