38 void GeoGridHDF5Writer::Write(GeoGrid& geoGrid)
39 {
40  try {
41  // Turn off the auto-printing when failure occurs so that we can
42  // handle the errors appropriately
43  Exception::dontPrint();
44 
45  H5File file(m_file_name, H5F_ACC_TRUNC);
46 
47  Group locations_group(file.createGroup("/Locations"));
48  WriteAttribute(geoGrid.size(), "size", locations_group);
49 
50  for (const auto& location : geoGrid) {
51  WriteLocation(*location, locations_group);
52  }
53 
54  WritePeople(file);
55 
56  m_persons_found.clear();
57 
58  } catch (const Exception& e) {
59  throw stride::util::Exception("There was an error writing the GeoGrid to the file: " +
60  e.getDetailMsg());
61  }
62 }

64 void GeoGridHDF5Writer::WriteLocation(Location& loc, Group& locsGroup)
65 {
67 
68  Group locGroup(locsGroup.createGroup("Location" + to_string(m_location_counter++)));
70  WriteAttribute(loc.GetID(), "id", locGroup);
71  WriteAttribute(loc.GetName(), "name", locGroup);
72  WriteAttribute(loc.GetProvince(), "province", locGroup);
73  WriteAttribute(loc.GetPopCount(), "pop_count", locGroup);
74  WriteAttribute(loc.GetCoordinate().get<0>(), "longitude", locGroup);
75  WriteAttribute(loc.GetCoordinate().get<1>(), "latitude", locGroup);
76 
77  auto commutes = loc.CRefOutgoingCommutes();
78  WriteCommutes(commutes, locGroup);
79 
80  Group cpGroup(locGroup.createGroup("ContactPools"));
81  for (Id type : IdList) {
82  for (auto pool : loc.RefPools(type)) {
83  WriteContactPool(*pool, type, cpGroup);
84  }
85  }
86  WriteAttribute(m_pool_counter, "size", cpGroup);
87  m_pool_counter = 0;
88 }

90 void GeoGridHDF5Writer::WriteCommutes(const vector<pair<Location*, double>>& commutes, Group& locGroup)
91 {
93  CompType commute_t = GetCompoundType<H5Commute>();
94 
96  DataSet dataset(locGroup.createDataSet("commutes", commute_t, CreateSpace(commutes.size())));
97  WriteAttribute(commutes.size(), "size", dataset);
98 
99  vector<H5Commute> comms(commutes.size());
100  for (unsigned long i = 0; i < commutes.size(); ++i) {
101  comms[i].to = commutes[i].first->GetID();
102  comms[i].proportion = commutes[i].second;
103  }
104 
105  dataset.write(comms.data(), commute_t);
106 }

108 void GeoGridHDF5Writer::WriteContactPool(const ContactPool& pool, Id type, Group& cpGroup)
109 {
110 
111  CompType pool_person_t = GetCompoundType<H5PoolPerson>();
112 
114  string label = "Pool" + to_string(m_pool_counter++);
115  DataSet dataset(cpGroup.createDataSet(label, pool_person_t, CreateSpace(pool.size())));
116  WriteAttribute(pool.GetId(), "id", dataset);
117  WriteAttribute(pool.size(), "size", dataset);
118  WriteAttribute(ToString(type), "type", dataset);
119 
120  vector<H5PoolPerson> people(pool.size());
121  for (unsigned long i = 0; i < pool.size(); ++i) {
122  people[i].id = pool[i]->GetId();
123  m_persons_found.insert(pool[i]);
124  }
125 
126  dataset.write(people.data(), pool_person_t);
127 }

129 void GeoGridHDF5Writer::WritePeople(Group& rootGroup)
130 {
131 
133  CompType person_t = GetCompoundType<H5Person>();
134 
136  DataSet dataset(rootGroup.createDataSet("People", person_t, CreateSpace(m_persons_found.size())));
137  WriteAttribute(m_persons_found.size(), "size", dataset);
138 
139  vector<H5Utils::H5Person> people(m_persons_found.size());
140  int y = 0;
141  for (auto p : m_persons_found) {
142  people[y].id = p->GetId();
143  people[y].age = p->GetAge();
144  people[y].Household = p->GetPoolId(Id::Household);
145  people[y].Daycare = p->GetPoolId(Id::Daycare);
146  people[y].PreSchool = p->GetPoolId(Id::PreSchool);
147  people[y].K12School = p->GetPoolId(Id::K12School);
148  people[y].College = p->GetPoolId(Id::College);
149  people[y].Workplace = p->GetPoolId(Id::Workplace);
150  people[y].PrimaryCommunity = p->GetPoolId(Id::PrimaryCommunity);
151  people[y].SecondaryCommunity = p->GetPoolId(Id::SecondaryCommunity);
152  y++;
153  }
154 
155  dataset.write(people.data(), person_t);
