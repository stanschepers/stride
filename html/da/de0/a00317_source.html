32 double distanceOnEarth(double lat1, double long1, double lat2, double long2)
33 {
34  // double a = std::pow((std::sin((lat2 - lat1) / 2) * M_PI) / 180, 2) + std::cos((lat1 * M_PI) / 180) *
35  // std::cos((lat2 * M_PI) / 180) * std::pow((std::sin((long2 - long1) / 2) * M_PI) / 180, 2); a = 2 *
36  // 6371000 * std::atan2(std::sqrt(a), std::sqrt(1 - a));
37  double a = std::acos(std::sin((lat1 * M_PI) / 180) * std::sin((lat2 * M_PI) / 180) +
38  std::cos((lat1 * M_PI) / 180) * std::cos((lat2 * M_PI) / 180) *
39  std::cos(((long2 - long1) * M_PI) / 180)) *
40  6731000;
41  return a;
42 }

50 QString makePresentablePercentage(double value)
51 {
52  // If -1 just return -1
53  if (value == -1) {

114 void MapController::SetShownInformation(const QString& sourceInformation)
115 {
116  // If a location is pinned, don't update the shown data
117  if (m_data_pinned) {
118  return;
119  }
120 
121  // Keep in member which is the last shown location
122  m_pinned_data = sourceInformation;
123 
124  // Tokenize the given information
125  std::istringstream buf(sourceInformation.toStdString());
126  std::istream_iterator<std::string> beg(buf), end;
127 
128  std::vector<std::string> tokens(beg, end);
129 
130  // Initialize the needed variables
131  std::string header;
132  std::map<std::string, std::map<std::string, std::map<unsigned int, double>>> info;
133 
134  // Update the shown info in the correct way
135  switch (tokens.size()) {
136  case 0: // No source: display nothing / clear
137  QMetaObject::invokeMethod(m_root, "emptyData");
138  return;
139  case 1: // Location:
140  // Find the right location and send the data to the GUI
141  for (const auto& location : m_geogrid) {
142  if (location->GetID() == std::stoul(tokens[0])) {
143  header = location->GetName();
144  info = location->GetContent()->epiOutput;
145  break;
146  }
147  }
148  break;
149  case 3: // Circle:
150  {
151  header = "Circle selection";
152  geopop::Location<EpiOutput> tempLoc(std::numeric_limits<unsigned int>::infinity(),
153  std::numeric_limits<unsigned int>::infinity(), nullptr,
154  geopop::Coordinate(std::stod(tokens[0]), std::stod(tokens[1])));
155  std::vector<const geopop::Location<EpiOutput>*> locations = m_geogrid.LocationsInRadius(tempLoc, std::stod(tokens[2]) / 1000);
156  // Calculate the total population of the selected area
157  double total_pop = 0;
158  for (auto const& location : locations) {
159  total_pop += location->GetContent()->pop_count;
160  }
161  // Calculate the total percentage
162  for (const std::string& ageBracket : stride::ageBrackets) {
163  for (const std::string& healthStatus : stride::healthStatuses) {
164  double total = 0;
165  for (auto const& location : locations) {
166  // Let the percentage count for the size the population is
167  total += location->GetContent()->epiOutput[ageBracket][healthStatus][m_day] *
168  (location->GetContent()->pop_count / total_pop);
169  }
170  info[ageBracket][healthStatus][m_day] = total;
171  }
172  }
173  break;
174  }
175  case 4: // Rectangle
176  {
177  header = "Rectangle selection";
178  std::set<const geopop::Location<EpiOutput>*> locations = m_geogrid.LocationsInBox(
179  std::stod(tokens[0]), std::stod(tokens[1]), std::stod(tokens[2]), std::stod(tokens[3]));
180  // Calculate the total population of the selected area
181  double total_pop = 0;
182  for (auto const& location : locations) {
183  total_pop += location->GetContent()->pop_count;
184  }
185  // Calculate the total percentage
186  for (const std::string& ageBracket : stride::ageBrackets) {
187  for (const std::string& healthStatus : stride::healthStatuses) {
188  double total = 0;
189  for (auto const& location : locations) {
190  // Let the percentage count for the size the population is
191  total += location->GetContent()->epiOutput[ageBracket][healthStatus][m_day] *
192  (location->GetContent()->pop_count / total_pop);
193  }
194  info[ageBracket][healthStatus][m_day] = total;
195  }
196  }
197  break;
198  }
199  default: header = "ERROR"; break;
200  }
201 
202  // Create a Qt variable to pass to QML
203  QVariantMap shownInfo;
204  for (const std::string& ageBracket : stride::ageBrackets) {
205  QVariantMap ageInfo;
206  for (const std::string& healthStatus : stride::healthStatuses) {
207  ageInfo[QString::fromStdString(healthStatus)] =
208  makePresentablePercentage(info[ageBracket][healthStatus][m_day]);
209  }
210  shownInfo[QString::fromStdString(ageBracket)] = QVariant(ageInfo);
211  }
212 
213  // Call the QML method to update the show info in the GUI
214  QMetaObject::invokeMethod(m_root, "setData", Q_ARG(QVariant, QString::fromStdString(header)),
215  Q_ARG(QVariant, shownInfo));
216 }
