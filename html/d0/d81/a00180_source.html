117 std::tuple<bool, unsigned int> ContactPool::SortMembers()
118 {
119  bool infectious_cases = false;
120  unsigned int num_cases = 0;
121 
122  for (size_t i_member = 0; i_member < m_index_immune; i_member++) {
123  // if immune, move to back
124  if (m_members[i_member]->GetHealth().IsImmune()) {
125  bool swapped = false;
126  unsigned int new_place = m_index_immune - 1;
127  m_index_immune--;
128  while (!swapped && new_place > i_member) {
129  if (m_members[new_place]->GetHealth().IsImmune()) {
130  m_index_immune--;
131  new_place--;
132  } else {
133  swap(m_members[i_member], m_members[new_place]);
134  swapped = true;
135  }
136  }
137  }
138  // else, if not susceptible, move to front
139  else if (!m_members[i_member]->GetHealth().IsSusceptible()) {
140  if (!infectious_cases && m_members[i_member]->GetHealth().IsInfectious()) {
141  infectious_cases = true;
142  }
143  if (i_member > num_cases) {
144  swap(m_members[i_member], m_members[num_cases]);
145  }
146  num_cases++;
147  }
148  }
149  return std::make_tuple(infectious_cases, num_cases);
150 }
