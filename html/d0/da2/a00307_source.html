28 void Generator<stride::ContactType::Id::Workplace>::Apply(GeoGrid& geoGrid, const GeoGridConfig& ggConfig)
29 {
30  // 1. active people count and the commuting people count are given
31  // 2. count the workplaces, either by poolparams or if given: the work distribution
32  // 3. count the working people at each location = #residents + #incoming commuters - #outgoing commuters
33  // 4. use that information for the distribution
34  // 5. assign each workplaces to a location
35 
36  const auto EmployeeCount = ggConfig.info.popcount_workplace;
37  auto WorkplacesCount =
38  static_cast<unsigned int>(ceil(EmployeeCount / static_cast<double>(ggConfig.people[Id::Workplace])));
39 
40  // if distribution file is present, make accurate estimation of WorkplacesCount
41  const auto distribution = ggConfig.param.work_distribution;
42  vector<double> workWeights;
43  if (!distribution.empty()) {
44  double PoolSize = 0;
45  for (auto entry : distribution) {
46  double ratio = std::get<0>(entry);
47  unsigned int minSize = std::get<1>(entry);
48  unsigned int maxSize = std::get<2>(entry);
49 
50  workWeights.emplace_back(ratio);
51  double avg = (minSize + maxSize) / 2.0;
52  PoolSize += avg * ratio;
53  }
54  const auto NewWorkplacesCount = static_cast<unsigned int>(ceil(EmployeeCount / PoolSize));
55  m_logger->trace("Number of workplaces through distribution file: " + to_string(NewWorkplacesCount) +
56  ", instead of: " + to_string(WorkplacesCount));
57  WorkplacesCount = NewWorkplacesCount;
58  }
59 
60  // = for each location #residents + #incoming commuting people - #outgoing commuting people
61  vector<double> weights;
62  for (const auto& loc : geoGrid) {
63  const double ActivePeopleCount =
64  (loc->GetPopCount() + loc->GetIncomingCommuteCount(ggConfig.param.fraction_workplace_commuters) -
65  loc->GetOutgoingCommuteCount(ggConfig.param.fraction_workplace_commuters) *
66  ggConfig.param.participation_workplace);
67 
68  const double weight = ActivePeopleCount / EmployeeCount;
69  AssertThrow(weight >= 0 && weight <= 1 && !std::isnan(weight), "Invalid weight: " + to_string(weight),
70  m_logger);
71  weights.push_back(weight);
72  }
73 
74  if (weights.empty()) {
75  // trng can't handle empty vectors
76  return;
77  }
78 
79  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
80  auto pop = geoGrid.GetPopulation();
81 
82  unsigned int entryIndex = 0;
83  int totalSize = 0;
84  for (auto i = 0U; i < WorkplacesCount; i++) {
85  const auto loc = geoGrid[dist()];
86 
87  // ----------------------------------------------
88  // If there's distribution, add pools using limit
89  // ----------------------------------------------
90  if (!distribution.empty()) {
91  auto workClass = m_rn_man.GetDiscreteGenerator(workWeights, 0U);
92  entryIndex = workClass();
93  auto minSize = std::get<1>(distribution[entryIndex]);
94  auto maxSize = std::get<2>(distribution[entryIndex]);
95 
96  auto genSize = m_rn_man.GetUniformIntGenerator(minSize, maxSize + 1U); // [a, b[
97  unsigned int s = genSize();
98  totalSize += s;
99  AddPools(*loc, pop, ggConfig, s);
100  } else {
101  AddPools(*loc, pop, ggConfig);
102  }
103  }
104 
105  if (!distribution.empty()) {
106  m_logger->trace("Total generated size in workplaces: " + to_string(totalSize));
107  m_logger->trace("Total number of employees: " + to_string(EmployeeCount));
108  }
109 }
