31 void PreSchoolGenerator::Apply(GeoGrid& geoGrid, const GeoGridConfig& geoGridConfig)
32 {
33  // 1. given the number of persons of school age, calculate number of schools; schools
34  // have 500 pupils on average
35  // 2. assign schools to a location by using a discrete distribution which reflects the
36  // relative number of pupils for that location; the relative number of pupils is set
37  // to the relative population w.r.t the total population.
38 
39  const auto pupilCount = geoGridConfig.popInfo.popcount_preschool;
40  const auto schoolCount =
41  static_cast<unsigned int>(ceil(pupilCount / static_cast<double>(geoGridConfig.pools.preschool_size)));
42 
43  vector<double> weights;
44  for (const auto& loc : geoGrid) {
45  weights.push_back(loc->GetPopFraction());
46  }
47 
48  if (weights.empty()) {
49  // trng can't handle empty vectors
50  return;
51  }
52 
53  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
54  auto pop = geoGrid.GetPopulation();
55 
56  for (auto i = 0U; i < schoolCount; i++) {
57  const auto loc = geoGrid[dist()];
58  AddPools(*loc, pop, geoGridConfig.pools.pools_per_preschool);
59  }
60 }
