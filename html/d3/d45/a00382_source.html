28 void Generator<stride::ContactType::Id::Workplace>::Apply(GeoGrid<Epidemiologic>& geoGrid,
29  const GeoGridConfig& ggConfig)
30 {
31  // 1. active people count and the commuting people count are given
32  // 2. count the workplaces, either by poolparams or if given: the work distribution
33  // 3. count the working people at each location = #residents + #incoming commuters - #outgoing commuters
34  // 4. use that information for the distribution
35  // 5. assign each workplaces to a location
36 
37  const auto EmployeeCount = ggConfig.info.popcount_workplace;
38  auto WorkplacesCount =
39  static_cast<unsigned int>(ceil(EmployeeCount / static_cast<double>(ggConfig.people[Id::Workplace])));
40 
41  // if distribution file is present, make accurate estimation of WorkplacesCount
42  const auto distribution = ggConfig.param.work_distribution;
43  vector<double> workWeights;
44  if (!distribution.empty()) {
45  double PoolSize = 0;
46  for (auto entry : distribution) {
47  double ratio = std::get<0>(entry);
48  unsigned int minSize = std::get<1>(entry);
49  unsigned int maxSize = std::get<2>(entry);
50 
51  workWeights.emplace_back(ratio);
52  double avg = (minSize + maxSize) / 2.0;
53  PoolSize += avg * ratio;
54  }
55  const auto NewWorkplacesCount = static_cast<unsigned int>(ceil(EmployeeCount / PoolSize));
56  m_logger->trace("Number of workplaces through distribution file: " + to_string(NewWorkplacesCount) +
57  ", instead of: " + to_string(WorkplacesCount));
58  WorkplacesCount = NewWorkplacesCount;
59  }
60 
61  // = for each location #residents + #incoming commuting people - #outgoing commuting people
62  vector<double> weights;
63  for (const auto& loc : geoGrid) {
64  const double ActivePeopleCount =
65  (loc->GetContent()->GetPopCount() +
66  loc->GetContent()->GetIncomingCommuteCount(ggConfig.param.fraction_workplace_commuters) -
67  loc->GetContent()->GetOutgoingCommuteCount(ggConfig.param.fraction_workplace_commuters) *
68  ggConfig.param.participation_workplace);
69 
70  const double weight = ActivePeopleCount / EmployeeCount;
71  AssertThrow(weight >= 0 && weight <= 1 && !std::isnan(weight), "Invalid weight: " + to_string(weight),
72  m_logger);
73  weights.push_back(weight);
74  }
75 
76  if (weights.empty()) {
77  // trng can't handle empty vectors
78  return;
79  }
80 
81  const auto dist = m_rn_man.GetDiscreteGenerator(weights, 0U);
82  auto pop = geoGrid.GetPopulation();
83 
84  unsigned int entryIndex = 0;
85  int totalSize = 0;
86  for (auto i = 0U; i < WorkplacesCount; i++) {
87  const auto loc = geoGrid[dist()];
88 
89  // ----------------------------------------------
90  // If there's distribution, add pools using limit
91  // ----------------------------------------------
92  if (!distribution.empty()) {
93  auto workClass = m_rn_man.GetDiscreteGenerator(workWeights, 0U);
94  entryIndex = workClass();
95  auto minSize = std::get<1>(distribution[entryIndex]);
96  auto maxSize = std::get<2>(distribution[entryIndex]);
97 
98  auto genSize = m_rn_man.GetUniformIntGenerator(minSize, maxSize + 1U); // [a, b[
99  unsigned int s = genSize();
100  totalSize += s;
101  AddPools(*loc, pop, ggConfig, s);
102  } else {
103  AddPools(*loc, pop, ggConfig);
104  }
105  }
106 
107  if (!distribution.empty()) {
108  m_logger->trace("Total generated size in workplaces: " + to_string(totalSize));
109  m_logger->trace("Total number of employees: " + to_string(EmployeeCount));
110  }
111 }
