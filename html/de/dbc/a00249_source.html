37 void GeoGridJSONReader::Read()
38 {
39  json data;
40 
41  try {
42  *(m_inputStream.get()) >> data;
43  data["locations"];
44  data["persons"];
45  } catch (const json::exception& error) {
46  throw runtime_error("An error occured while parsing JSON. Please make sure valid JSON is provided.");
47  }
48 
49  json dummyJSON;
50 
51  auto& geoGrid = m_population->RefGeoGrid();
52  auto& locations = data["locations"].is_string() ? dummyJSON : data["locations"];
53  auto& people = data["persons"].is_string() ? dummyJSON : data["persons"];
54 
55  for (const auto& person : people) {
56  auto stridePerson = ParsePerson(person);
57  m_people[stridePerson->GetId()] = stridePerson;
58  }
59 
60  for (const auto& location : locations) {
61  auto strideLocation = ParseLocation(location);
62  geoGrid.AddLocation(move(strideLocation));
63  }
64 
65  AddCommutes(geoGrid);
66  m_commutes.clear();
67  m_people.clear();
68 }

101 std::shared_ptr<Location<Epidemiologic>> GeoGridJSONReader::ParseLocation(const json& location)
102 {
103  const auto id = ParseValue<unsigned int>(location["id"]);
104  const auto name = ParseValue<string>(location["name"]);
105  const auto province = ParseValue<unsigned int>(location["province"]);
106  const auto population = ParseValue<unsigned int>(location["population"]);
107  const auto coordinate = ParseCoordinate(location["coordinate"]);
108 
109  auto locationPtr = make_shared<Location<Epidemiologic>>(id, province, coordinate, name, population);
110 
111  if (location.find("contactPools") != location.end() && location["contactPools"].is_array()) {
112  for (const auto& contactPool : location["contactPools"]) {
113  const auto type = ParseValue<string>(contactPool["class"]);
114 
115  static const map<string, Id> types = {{"K12School", Id::K12School},
116  {"Daycare", Id::Daycare},
117  {"PreSchool", Id::PreSchool},
118  {"PrimaryCommunity", Id::PrimaryCommunity},
119  {"SecondaryCommunity", Id::SecondaryCommunity},
120  {"College", Id::College},
121  {"Household", Id::Household},
122  {"Workplace", Id::Workplace}};
123 
124  const auto typeId = types.at(type);
125 
126  if (contactPool.find("pools") != contactPool.end() && contactPool["pools"].is_array()) {
127  for (const auto& pool : contactPool["pools"]) {
128  ParseContactPool(locationPtr, pool, typeId);
129  }
130  }
131  }
132  }
133 
134  if (location.find("commutes") != location.end() && location["commutes"].is_array()) {
135  for (const auto& commute : location["commutes"]) {
136  const auto to = ParseValue<unsigned int>(commute["to"]);
137  const auto proportion = ParseValue<double>(commute["proportion"]);
138  m_commutes.emplace_back(id, to, proportion);
139  }
140  }
141 
142  return locationPtr;
143 }

152 void GeoGridJSONReader::ParseContactPool(const shared_ptr<Location<Epidemiologic>>& location, const json& contactPool,
153  Id type)
154 {
155  // Don't use the id of the ContactPool but the let the Population create an id
156  auto contactPoolPtr = m_population->RefPoolSys().CreateContactPool(type);
157  location->GetContent()->RefPools(type).emplace_back(contactPoolPtr);
158 
159  if (contactPool.find("people") != contactPool.end() && contactPool["people"].is_array()) {
160  for (const auto& id : contactPool["people"]) {
161  const auto person_id = ParseValue<unsigned int>(id);
162  const auto& person = m_people.at(person_id);
163  contactPoolPtr->AddMember(person);
164  // Update original pool id with new pool id used in the population
165  person->SetPoolId(type, contactPoolPtr->GetId());
166  }
167  }
168 }
