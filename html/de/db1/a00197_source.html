110 void GeoGridHDF5Reader::ReadLocation(const H5::Group& location, GeoGrid& grid)
111 {
113  unsigned int id, pop_count, province;

115  string name;
116 
117  ReadAttribute(id, "id", location);
118  ReadAttribute(pop_count, "pop_count", location);
119  ReadAttribute(province, "province", location);
120  ReadAttribute(longitude, "longitude", location);
121  ReadAttribute(latitude, "latitude", location);
122  ReadAttribute(name, "name", location);
123 
124  auto locationPtr = make_shared<Location>(id, province, Coordinate(longitude, latitude), name, pop_count);
125 
127  Group contact_pools(location.openGroup("ContactPools"));
128  unsigned int contact_pool_count;
129

146  commutes.read(commute_data.data(), GetCompoundType<H5Commute>());
147

157 void GeoGridHDF5Reader::ReadContactPool(const H5::DataSet& pool, const shared_ptr<Location>& location)
158 {
159  static const map<string, Id> types = {{"K12School", Id::K12School},
160  {"PrimaryCommunity", Id::PrimaryCommunity},
161  {"SecondaryCommunity", Id::SecondaryCommunity},
162  {"College", Id::College},
163  {"Household", Id::Household},
164  {"Daycare", Id::Daycare},
165  {"PreSchool", Id::PreSchool},
166  {"Workplace", Id::Workplace}};
167 
168  size_t size;
169  string type;
170 
171  ReadAttribute(size, "size", pool);
172  ReadAttribute(type, "type", pool);
173 
174  Id type_id = types.at(type);
175  vector<H5PoolPerson> pool_data(size);
176  pool.read(pool_data.data(), GetCompoundType<H5PoolPerson>());
177 
178  // Don't use the id of the ContactPool but the let the Population create an id
179  auto contactPoolPtr = m_population->RefPoolSys().CreateContactPool(type_id);
180  location->RefPools(type_id).emplace_back(contactPoolPtr);
181 
182  if (size > 0) {
183  for (auto p : pool_data) {
184  const auto person_id = p.id;
185  const auto& person = m_people.at(person_id);
186  contactPoolPtr->AddMember(person);
187  // Update original pool id with new pool id used in the population
188  person->SetPoolId(type_id, contactPoolPtr->GetId());
189  }
190  }
191 }
