53 nlohmann::json GeoGridJSONWriter::WriteLocation(const Location& location)
54 {
55  json locationJSON;
56 
57  locationJSON["id"] = location.GetID();
58  locationJSON["name"] = location.GetName();
59  locationJSON["province"] = location.GetProvince();
60  locationJSON["population"] = location.GetPopCount();
61  locationJSON["coordinate"] = WriteCoordinate(location.GetCoordinate());
62  locationJSON["contactPools"] = json::array();
63  locationJSON["commutes"] = json::array();
64 
65  for (const auto& type : IdList) {
66 
67  json contactPoolJSON;
68  contactPoolJSON["class"] = ToString(type);
69  contactPoolJSON["pools"] = json::array();
70 
71  for (const auto& pool : location.CRefPools(type)) {
72  contactPoolJSON["pools"].push_back(WriteContactPool(pool));
73  }
74  locationJSON["contactPools"].push_back(contactPoolJSON);
75  }
76 
77  for (const auto& commute : location.CRefOutgoingCommutes()) {
78  json commuteJSON;
79  commuteJSON["to"] = get<0>(commute)->GetID();
80  commuteJSON["proportion"] = get<1>(commute);
81  locationJSON["commutes"].push_back(commuteJSON);
82  }
83 
84  return locationJSON;
