60 int Epidemiologic::GetIncomingCommuteCount(double fractionCommuters) const
61 {
62  double value = 0;
63  for (const auto& locProportion : m_inCommutes) {
64  // locProportion.second of the people in locProportion.first are commuting to this
65  value += locProportion.second * (fractionCommuters * (double)locProportion.first->GetPopCount());
66  }
67  return static_cast<int>(floor(value));
68 }

105 std::map<std::string, std::map<std::string, double>> const Epidemiologic::GenerateEpiOutput()
106 {
107  // Get the total amount of members in the Epidemiologic
108  unsigned int total_members = this->GetMemberCount();
109 
110  // Initialize the map
111  std::map<std::string, std::map<std::string, double>> epiOutput;
112  for (const string& ageBracket : stride::ageBrackets) {
113  for (const string& healthStatus : stride::healthStatuses) {
114  epiOutput[ageBracket][healthStatus] = 0;
115  }
116  }
117 
118  if (total_members != 0) {
119  // Iterate over all Household contactpools in the Epidemiologic and calculate the total
120  for (stride::ContactPool* pool : this->CRefPools(stride::ContactType::Id::Household)) {
121  std::map<std::string, std::map<std::string, unsigned int>> epiOutput_pool =
122  pool->GenerateEpiOutput();
123  for (const string& ageBracket : stride::ageBrackets) {
124  for (const string& healthStatus : stride::healthStatuses) {
125  epiOutput[ageBracket][healthStatus] +=
126  double(epiOutput_pool[ageBracket][healthStatus]);
127  }
128  }
129  }
130  // Divide the totals by the total in the age bracket to get the epi-output
131  for (const string& ageBracket : stride::ageBrackets) {
132  if (epiOutput[ageBracket]["Total"] == 0) {
133  continue;
134  }
135  for (const string& healthStatus : stride::healthStatuses) {
136  if (healthStatus == "Total") { // If the total health status,
137  continue;
138  } else {
139  epiOutput[ageBracket][healthStatus] /= epiOutput[ageBracket]["Total"];
140  }
141  }
142  }
143  // Divide the total of each age bracket by the total pop count
144  for (const string& ageBracket : stride::ageBrackets) {
145  epiOutput[ageBracket]["Total"] /= double(total_members);
146  }
147  }
148 
149  return epiOutput;
150 }
